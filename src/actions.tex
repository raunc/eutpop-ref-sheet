\actionHeadingNoSpace{Event \normal{(p.~12)}}
\begin{itemize}
	\item Must be taken exactly once per Round
\end{itemize}
\begin{enumerate}
	\item \strong{Select} a face-up Event, \strong{collect} \ducats on the card and \strong{move} Round Status marker	\item Add 2 \ducats to remaining face-up Events
	\item Resolve effects\zsavepos{arrowEventEffectsStart}
	\item \hspacezero\zsavepos{arrowEventSymbolsStart}Resolve secondary effects, left to right
	\item Replace Ruler (opt. if other PR's Turn)
	\item If everyone has taken an \action{Event} and unpicked Event remains
	\begin{itemize}
		\item Auto-resolve secondary effects
		\item Ruler may be appointed for 2\adminpower
	\end{itemize}
	\item Flip a face-down Event, if any remain
\end{enumerate}

\begin{dynamiccontents*}{frameEventGuidelines}\begin{eubox}{\fhEventGuidelines}
	\begin{multicols*}{2}
		\subsubsection*{\zsavepos{arrowEventEffectsEnd}General Event guidelines \normal{(p.~40)}}
		\begin{itemize}
			\item In lists, word "or" is inclusive
			\item Ignore effects if conditions are not met
			\item Pick only one of A/B options (p.~12)
			\item When cannot pay required cost for one of A/B options and can pay for the other one, then must pick the other one
			\item If an effect/choice affects >1 PR, resolve starting from Active player
			\item PR that "gains/loses" makes the choice
			\item NPRs say "yes", when offered Alliance or when they "may" do something
			\item Word "you" refers to Active player or player making A/B choice
		\end{itemize}
		
		\subsubsection*{Standard Event effects \normal{(p.~40)}}
		\smallheading{Gain/place Influence}
		\begin{itemize}
			\item 1+ Province in Area must be Owned by NPR/Vassal
			\item May replace \influence if all target Areas are full
		\end{itemize}
		\smallheading{Gain Alliance}
		\begin{itemize}
			\item Target must be Independent \conj{and} at Peace
			\item If target is NPR \ally of an Opponent
			\begin{itemize}
				\item Your \influence must be ≥ Opponent's \influence
			\end{itemize}
			\item If out of \alliances, may remove one (except \activeally) with no penalties
		\end{itemize}
		\smallheading{Gain Royal Marriage}
		\begin{itemize}
			\item Target must be eligible
			\item If out of \marriages, may remove one (except \disputedsuccession)
		\end{itemize}
		\smallheading{Gain \claim}
		\begin{itemize}
			\item May have only 1 per Area
		\end{itemize}
		\smallheading{Gain/place \town or \vassal}
		\begin{itemize}
			\item Target Prov. may not have Opp.'s \town/\vassal
		\end{itemize}
		\smallheading{Gain Merchant}
		\begin{itemize}
			\item Only if you have 2 merchants
		\end{itemize}
		\smallheading{Gain Military Unit}
		\begin{itemize}
			\item Must take Reg. Units from Available \manpower
			\item May not trigger a Battle
		\end{itemize}
		\smallheading{Gain Action}
		\begin{itemize}
			\item Must be taken immediately
		\end{itemize}
		\smallheading{Develop a Small \town into a Large \town}
		\begin{itemize}
			\item \town must be Owned
		\end{itemize}
		\smallheading{\action{Research} Idea}
		\begin{itemize}
			\item Score \prestige as normal (Active PR first)
			\item If the Idea is not on display
			\begin{itemize}
				\item Replace unpicked non-Basic Idea of the same type
				\item If no unpicked non-Basic Idea, add the Idea to a new row
			\end{itemize}
		\end{itemize}
		\smallheading{Placing DNPR tokens}
		\begin{itemize}
			\item Use the token with C as Capital (underlined in Event text)
			\item If target Province has \town/\vassal, then \dnpr goes underneath
			\begin{itemize}
				\item May not be placed on PR's Core Prov.
			\end{itemize}
			\item If first \dnpr on a Dist. Cont., add Tr. Cards\zsavepos{arrowdistantTradeStartOne}
		\end{itemize}
		\smallheading{Terminate/remove Alliance}
		\begin{itemize}
			\item If \activeally, lose Allied Units = ½ of NPR's pre-War Strength
			\item Enemy places War tokens on former \activeally
		\end{itemize}
		\smallheading{Reject Ruler}
		\begin{itemize}
			\item Discard current Ruler and Event's ruler once Event is resolved
		\end{itemize}
		\smallheading{Area Leaves HRE}
		\begin{itemize}
			\item Loss of \authority is accounted for in the text
		\end{itemize}
		\smallheading{Wars triggered by Events}
		\begin{itemize}
			\item If not \action{Declare War} Action, do steps 3-7
			\item DoW restrictions apply
			\item PR that has Passed cannot Declare War
		\end{itemize}
	\end{multicols*}
\end{eubox}\end{dynamiccontents*}

		\begin{itemize}
			\item Place a Pirate into maritime Trade Node
			\begin{itemize}
				\item Trade Node must have 1+ Merchant, if possible
			\end{itemize}
			\item \strong{Auto-resolution:} Node with most Merchants. Tie breakers: 1.~Exp. Node; 2.~random
		\end{itemize}
		\smallheading{Cardinal Dies \eventIcon{cardinal_dies}}
		\begin{itemize}
			\item Remove any \cardinal, except the Roma \cardinal
			\item \strong{Auto-resolution:}
			\begin{itemize}
				\item Remove rightmost \cardinal from the PR with most \cardinal
				\item If tied, remove rightmost \cardinal of those
			\end{itemize}
		\end{itemize}
		\smallheading{Character Mortality \eventIcon{mortality}}
		\begin{itemize}
			\item All matching characters gain an \illhealth
			\item After the second \illhealth, the character dies
		\end{itemize}
		\smallheading{Unrest/Rebellion \eventIcon{rebellion}}
		\begin{itemize}
			\item All PRs roll Rebel Dice
			\item You roll max 1 Rebel Die
		\end{itemize}
		\smallheading{Attrition \eventIcon{attrition}}
		\begin{itemize}
			\item All PRs at War, except you, lose 1 for every 4 Deployed Land Units as Casualties
		\end{itemize}
		\smallheading{Lost at Sea \eventIcon{lost_at_sea}}
		\begin{itemize}
			\item Remove 1 Ship of each Opponent not Adjacent to a Friendly Port
			\item \strong{Auto-resol.:} PRs choose for themselves
		\end{itemize}
		\smallheading{DNPR Expansion \eventIcon{dnpr_expansion}}
		\begin{itemize}
			\item Pick a DNPR per matching color (not \activeally)
			\item For each picked DNPR
			\begin{itemize}
				\item Place \dnpr on an Adj. Prov. that is not
				\begin{itemize}
					\item Capital, unless NPR's last Province
					\item Occupied
					\item Owned by the DNPR's Ally
					\item Core Province of a PR
				\end{itemize}
				\item If DNPR has a Port facing any Main Map Sea Zone without */\textdagger, \conj{and} it is Age II or later
				\begin{itemize}
					\item May instead place the \dnpr on any vacant coastal Territory
					\item \hspacezero\zsavepos{arrowDistantTradeStartThree}If first \dnpr on a Distant Continent, add Trade Cards
					\end{itemize}
				\item If \dnpr was placed on a PR \town/\vassal, then
				\begin{itemize}
					\item \dnpr goes underneath the \town/\vassal
					\item Add \unrest to the \town/\vassal
				\end{itemize}
				\item If \dnpr was placed on a PR's NPR \ally
				\begin{itemize}
					\item That PR may place a CB token on the DNPR's Capital
				\end{itemize}
			\end{itemize}
			\item \strong{Auto-resolution:}
			\begin{itemize}
				\item All DNPRs with matching colors target an Adjacent NPR Province. Priority:
				\begin{enumerate}
					\item In Areas where they Own Provinces
					\item In Areas Adjacent by Land
					\item Owned by NPR with lowest Str.
					\item \az
				\end{enumerate}
			\end{itemize}
		\end{itemize}
		\smallheading{Spread of Religious Ideas \eventIcon{spread_religion}}
		\begin{itemize}
			\item Find clusters of Protes. and Counter-Ref. (bodies of Areas connected by Land)
			\item For each cluster, place \infectiousfaith in an Adjacent Area (except Muslim or Orthodox), \conj{but}
			\begin{itemize}
				\item Max 4 in total
				\item If 2 of one type already placed, then must place the other type if possible
				\item May not replace \infectiousfaith placed this Turn
			\end{itemize}
			\item If no clusters, then place Diverse Faiths tokens in any 2 Catholic Areas not Adjacent to Sea Zones marked with \textdagger
			\item \strong{Auto-resolution:}
			\begin{itemize}
				\item Start with Protestant
				\item Alternate between Prot. and Ctr-Ref.
				\item Area selection priority: 1.~Areas with \towns; 2.~highest Area Tax Value; 3.~\az
				\item If placing Diverse Faith, go \az
			\end{itemize}
		\end{itemize}
		\smallheading{Spread of the Revolution \eventIcon{spread_revolution}}
		\begin{itemize}
			\item Find clusters of \revolution

			\begin{itemize}
				\item Place \revolution in an Adjacent Area
			\end{itemize}
			\item If no \revolution, place \revolution in 2 PR Capital Areas
			\item If only 1 cluster, then also add a \revolution in an Area not Adjacent to this cluster
			\item Place on top of existing Religion tokens
			\item If out of \revolution, then may take \revolution from board
			\item \strong{Auto-resolution:}
			\begin{itemize}
				\item Area selection priority: same as \infectiousfaith
				\item If no existing \revolution, then select targets by: 1.~lowest \stability; 2.~random
			\end{itemize}
		\end{itemize}
		\smallheading{Activate Power Struggle (adv. rules) \eventIcon{power_struggle}}
		\begin{itemize}

			\item Place \battlegroundareas on Areas named on the card

\actionHeading{Player-to-Player Diplomacy \normal{(p.~12)}}
\begin{itemize}
	\item May interact with one player per Action
	\item May not do \strong{Monetary Support} and \strong{Buy/Sell Provinces} on the same turn
	\item If one of the PRs is at War, then may only do \strong{Monetary Support}
\end{itemize}
\smallheading{Marriage}
\begin{itemize}
	\item Active Player pays 1\diplopower
	\item Place \marriage tokens on both Capitals
	\item Both may draw an Action card of a type chosen by Active Player
\end{itemize}
\smallheading{Alliance}
\begin{itemize}
	\item Active player pays 1\diplopower
	\item Place \alliance tokens on both Capitals
\end{itemize}
\smallheading{Monetary Support}
\begin{itemize}
	\item Active player pays \diplopower = exchanged \ducats~/ 10
\end{itemize}
\smallheading{Buy/sell Provinces}
\begin{itemize}
	\item May only sell Own Provinces
	\item Buyer must have \claim in those Areas
	\item Price: 3-15\ducats per Province
	\item Both pay 1\adminpower
	\item PR selling Core Provinces
	\begin{itemize}
		\item Loses \prestige = 2× Tax Value
	\end{itemize}
\end{itemize}
\smallheading{Optional Rule 1: Secret Negotiations}
\begin{itemize}
	\item Pay 1\diplopower for a private 3-minute conference
	\item All other communication must be public
\end{itemize}

\actionHeading{Research Idea \normal{(X\monarchpower) (p.~12)}}
\begin{itemize}
	\item Mark it with a Tag
	\item Apply immediate effects
	\item Score \p2
	\item If ≤2 other PRs tagged it, they score \p1
	\item If you \action{Research} another \strong{Government Form} Idea while having one (p.~21)
	\begin{itemize}
		\item Lose the previous one (no \prestige loss)
		\item Score \prestige as normal for the new one
	\end{itemize}
	\item If \authority ≥ 3, \strong{Imperial Subjects} may not \action{Research} Government Form Ideas (p.~44)
\end{itemize}

\actionHeading{Change State Religion \normal{(p.~13)}}
\begin{itemize}
	\item Only from Age II onwards
	\item If HRE has official Rel., \strong{Emp.} may only change to HRE Rel. (adv. rules) (p.~45)
	\item Do one of the following
\end{itemize}
\smallheading{Convert between Catholic and Protestant}
\begin{itemize}
	\item Change State Rel. to Catholic/Protestant
	\item Lose \p2 \conj{and} 1\stability
	\item Lose 1 \marriage \conj{and} total of 5 \influence from Areas of abandoned Religion
	\item Place removed \influence to Areas of new Rel.
	\item In each of your Areas of abandoned Rel.
	\begin{itemize}
		\item Gain 1 \unrest
		\item Roll Rebel Dice
	\end{itemize}
\end{itemize}
\smallheading{Adopt Counter-Reformed Catholicism}
\begin{itemize}
	\item Allowed only when
	\begin{itemize}
		\item 1+ Counter-Ref. token is in play, \conj{and}
		\item You are Catholic
	\end{itemize}
	\item Pay 3\adminpower
	\item Change State Religion to Counter-Reformed
	\item Place Ctr-Ref. token to your Capital Area
\end{itemize}
